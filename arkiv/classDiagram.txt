    %% Backend Models
    class User {
        -String id
        -String email
        -String password
        -String role (admin|customer)
        -String fullname
        -String street
        -String zipCode
        -String city
        -String country
        -Date createdAt
        +login()
        +register()
    }

    class Product {
        -String id
        -String artist
        -String title
        -String category
        -String subcategory
        -Number price
        -String image
        -String description
        -Number releaseYear
        -Number stock
        +getProduct()
        +updateProduct()
        +deleteProduct()
    }

    class Order {
        -String id
        -String userId
        -OrderItem[] items
        -Number totalPrice
        -String status (pending|shipped|delivered)
        -Object shippingAddress
        -Date createdAt
        -Date shippedAt
        +createOrder()
        +updateStatus()
        +getOrderDetails()
    }

    class OrderItem {
        -String productId
        -Number quantity
        -Number price
    }

    %% Frontend Contexts
    class AuthContext {
        -User user
        -Boolean isAuthenticated
        -Boolean isAdmin
        +login(email, password)
        +registerUser(fullname, email, password, address)
        +registerAdmin(fullname, email, password)
        +logout()
    }

    class CartContext {
        -CartItem[] cartItems
        +addToCart(product)
        +removeFromCart(productId)
        +updateQuantity(productId, quantity)
        +clearCart()
        +getTotalPrice()
    }

    class CartItem {
        -String id
        -Number quantity
        -Number price
    }
